Usage

<script src="https://cdn.runwayml.com/characters.js"></script>
<script>
  const widget = RunwayCharacters.init({
    apiKey: 'pk_live_xxxxx',
    avatarId: 'my-avatar-id',
    onReady: () => console.log('Ready'),
    onError: (err) => console.error(err),
    onEnd: () => console.log('Session ended')
  });
</script>
